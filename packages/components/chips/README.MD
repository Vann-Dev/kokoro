# Usage

**Typescript**
```tsx
import { Chips } from "@kokoro-ui/chips";

const app = () => {
    return (
        <Chips
            className="border-secondary"
            variant="outline"
        >
            Label
        </Chips>
    )
};

export default app;
```

**Javascript**
```jsx
const { Chips } = require("@kokoro-ui/chips");

const app = () => {
    return (
        <Chips
            className="border-secondary"
            variant="outline"
        >
            Label
        </Chips>
    )
};

export default app;
```
