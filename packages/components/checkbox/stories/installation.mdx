# Usage

**Typescript**
```tsx
import { Checkbox } from "@kokoro-ui/checkbox";

const app = () => {
    return (
        <Checkbox
            onCheck={(isChecked: boolean) => { 
                console.log(isChecked); 
            }}
            selected
        >
            <div className="text-white">
                Checkbox Checked
            </div>
        </Checkbox>
    )
};

export default app;
```

**Javascript**
```jsx
const { checkbox } = require("@kokoro-ui/checkbox");

const app = () => {
    return (
        <Checkbox
            onCheck={(isChecked) => {
                console.log(isChecked);
            }}
            selected
        >
            <div className="text-white">
                Checkbox Checked
            </div>
        </Checkbox>
    )
};

export default app;
```
