# Usage

**Typescript**
```tsx
import { Badge } from "@kokoro-ui/badge";

const app = () => {
    return (
        <Badge
            badgeContent=""
            className="bg-green-500 border-2 border-secondary"
            position="bottom-right"
        >
            <Avatar
                alt="Jeanice"
                name="JC"
                src="https://i.pravatar.cc/300?u=4nfawf369026704d"
            />
        </Badge>
    )
};

export default app;
```
